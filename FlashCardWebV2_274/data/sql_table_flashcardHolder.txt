CREATE TABLE `flashcardholder` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `name` varchar(255) COLLATE utf8_unicode_ci NOT NULL,
  `fc_id` varchar(255) COLLATE utf8_unicode_ci DEFAULT NULL,
  `create_date` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `number_of_quiz_times` int(11) DEFAULT '0',
  `pass_the_quiz_times` int(11) DEFAULT '0',
  `update_for_quiz_date` timestamp NOT NULL DEFAULT '0000-00-00 00:00:00',
  PRIMARY KEY (`id`),
  KEY `hc_fc_fk` (`fc_id`),
  CONSTRAINT `hc_fc_fk` FOREIGN KEY (`fc_id`) REFERENCES `flashcard` (`id`) ON DELETE SET NULL ON UPDATE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=3 DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci
